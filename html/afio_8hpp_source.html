04634         BOOST_AFIO_ROUND(0)  // prefetch = min 8, bit test and xor = min 16, total = 24

04648     result_type operator()(result_type ecc, const char *buffer, size_t length) const BOOST_NOEXCEPT
04649     {
04650       const unit_type *_buffer = (const unit_type *) buffer;
04651       //#pragma omp parallel for reduction(^:ecc)
04652       for(size_t i = 0; i < length; i += sizeof(unit_type))
04653       {
04654         unit_type c = _buffer[i / sizeof(unit_type)];  // min 1 cycle
04655         if(!c)                                         // min 1 cycle
04656           continue;
04657         char bitset[bits_per_byte * sizeof(unit_type)];
04658         result_type prefetch[bits_per_byte * sizeof(unit_type)];
04659         // Most compilers will roll this out
04660         for(size_t n = 0; n < bits_per_byte * sizeof(unit_type); n++)  // min 16 cycles
04661         {
04662           bitset[n] = !!(c & ((unit_type) 1 << n));
04663           prefetch[n] = ecc_table[i * bits_per_byte + n];  // min 8 cycles
04664         }
04665         result_type localecc = 0;
04666         for(size_t n = 0; n < bits_per_byte * sizeof(unit_type); n++)
04667         {
04668           if(bitset[n])               // min 8 cycles
04669             localecc ^= prefetch[n];  // min 8 cycles
04670         }
04671         ecc ^= localecc;  // min 1 cycle. Total cycles = min 43 cycles/byte
04672       }
04673       return ecc;
04674     }

04677     result_type find_bad_bit(result_type good_ecc, result_type bad_ecc) const BOOST_NOEXCEPT
04678     {
04679       result_type length = blocksize * bits_per_byte, eccdiff = good_ecc ^ bad_ecc;
04680       if(_is_single_bit_set(eccdiff))
04681         return (result_type) -1;
04682       for(result_type i = 0, b = 1; i < length; i++, b++)
04683       {
04684         // Skip parity bits
04685         while(_is_single_bit_set(b))
04686           b++;
04687         if(b == eccdiff)
04688           return i;
04689       }
04690       return (result_type) -1;
04691     }

04700     verify_status verify(char *buffer, result_type good_ecc) const BOOST_NOEXCEPT
04701     {
04702       result_type this_ecc = (*this)(0, buffer);
04703       if(this_ecc == good_ecc)
04704         return verify_status::okay;  // no errors
04705       result_type badbit = find_bad_bit(good_ecc, this_ecc);
04706       if((result_type) -1 == badbit)
04707         return verify_status::corrupt;  // parity corrupt?
04708       buffer[badbit / bits_per_byte] ^= (unsigned char) ecc_twospowers[badbit % bits_per_byte];
04709       this_ecc = (*this)(0, buffer);
04710       if(this_ecc == good_ecc)
04711         return healed;  // error healed
04712       // Put the bit back
04713       buffer[badbit / bits_per_byte] ^= (unsigned char) ecc_twospowers[badbit % bits_per_byte];
04714       return verify_status::corrupt;  // more than one bit was corrupt
04715     }
